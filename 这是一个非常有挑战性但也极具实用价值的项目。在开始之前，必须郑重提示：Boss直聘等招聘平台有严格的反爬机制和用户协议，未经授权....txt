一、 项目大纲：Boss直聘高质量岗位智能聚合服务

二、 AI 辅助开发规范 (Harness & Best Practices)

在使用 AI 开发时，不要直接说“帮我写个爬虫”，而应遵循以下工程化规范：

1.  Context First (上下文优先): 每次对话前提供 tech-stack.md, schema.prisma, project-rules.md。
2.  Atomic Tasks (原子化任务): 将大需求拆解为单一职责的小任务（如：“创建 Job 表的 Prisma Schema” 而非 “做个数据库”）。
3.  TDD/BDD 导向: 要求 AI 先写测试用例或接口契约，再写实现。
4.  Structured Output (结构化输出): 强制 AI 返回 JSON/Zod Schema，而非自由文本。
5.  Safety Harness (安全护栏): 在 System Prompt 中明确禁止生成违反 ToS 的代码，要求使用官方 API 或浏览器扩展方式。

一、 Extension Popup 完整实现

Popup 是用户与插件交互的唯一窗口，需保持轻量、状态反馈即时。

 extension/src/popup.tsx (使用 Preact/React 构建)

import { useState, useEffect } from 'preact/hooks';

type SyncStatus = 'idle' | 'extracting' | 'syncing' | 'success' | 'error';

export function Popup() {
  const [status, setStatus] = useState<SyncStatus>('idle');
  const [count, setCount] = useState(0);
  const [errorMsg, setErrorMsg] = useState('');

  // 检查 Next.js 本地服务是否在线
  useEffect(() => {
    fetch('http://localhost:3000/api/health')
      .then(() => setStatus('idle'))
      .catch(() => {
        setStatus('error');
        setErrorMsg('未检测到 JobRadar 本地服务，请先启动 Next.js 项目');
      });
  }, []);

  const handleExtractAndSync = async () => {
    setStatus('extracting');
    setCount(0);
    setErrorMsg('');

    try {
      // 向 Content Script 发送消息
      const [tab] = await chrome.tabs.query({ active: true, currentWindow: true });
      
      if (!tab.id || !tab.url?.includes('zhipin.com')) {
        throw new Error('请在 Boss直聘 职位列表页使用此工具');
      }

      const response = await chrome.tabs.sendMessage(tab.id, { action: 'EXTRACT_AND_SYNC' });
      
      if (response?.success) {
        setCount(response.count);
        setStatus('success');
      } else {
        throw new Error(response?.error || '提取失败');
      }
    } catch (err: any) {
      setStatus('error');
      setErrorMsg(err.message);
    }
  };

  return (
    <div class="w-80 p-4 space-y-4 font-sans">
      <h1 class="text-lg font-bold text-gray-800"> JobRadar Assistant</h1>
      
      <div class="text-sm text-gray-600 bg-gray-50 p-3 rounded-md">
        {status === 'idle' && '✅ 本地服务已连接，准备就绪'}
        {status === 'extracting' && '⏳ 正在提取当前页面岗位...'}
        {status === 'syncing' && ' 正在同步至本地数据库...'}
        {status === 'success' && `✅ 成功同步 {count} 个岗位！AI 正在后台分析中`}
        {status === 'error' && `❌ {errorMsg}`}
      </div>

      <button
        onClick={handleExtractAndSync}
        disabled={status === 'extracting' || status === 'syncing' || status === 'error'}
        class={`w-full py-2 px-4 rounded-md text-white font-medium transition-colors {
          status === 'error' 
            ? 'bg-gray-400 cursor-not-allowed' 
            : 'bg-blue-600 hover:bg-blue-700 active:bg-blue-800'
        }`}
      >
        {status === 'extracting' || status === 'syncing' ? '处理中...' : '提取并同步当前页面'}
      </button>

      <p class="text-xs text-gray-400 text-center">
        仅在当前页面手动触发 · 数据仅存储于本地
      </p>
    </div>
  );
}

 3. 后台消费 Worker (Cron / 长轮询)

创建一个独立的 API 路由作为 Worker 入口，可通过 Vercel Cron / node-cron / 甚至前端定时 fetch 触发：

// app/api/workers/ai-analyze/route.ts
import { analyzeJob } from '@/services/ai-job-analyzer';
import { prisma } from '@/lib/prisma';

const BATCH_SIZE = 5; // 每批处理5个，控制并发
const MAX_RETRIES = 3;

export async function POST() {
  // 简单锁机制，防止多实例重复消费
  const lockKey = 'ai_worker_lock';
  // ... 省略分布式锁实现，单机可用内存变量或DB行锁
  
  try {
    // 1. 获取待处理任务
    const tasks = await prisma.aiAnalysisTask.findMany({
      where: { 
        status: 'pending',
        retryCount: { lt: MAX_RETRIES }
      },
      orderBy: { createdAt: 'asc' },
      take: BATCH_SIZE
    });

    if (tasks.length === 0) return Response.json({ processed: 0 });

    // 2. 标记为处理中（防重复）
    await prisma.aiAnalysisTask.updateMany({
      where: { id: { in: tasks.map(t => t.id) } },
      data: { status: 'processing' }
    });

    // 3. 并行调用 AI
    const results = await Promise.allSettled(
      tasks.map(async (task) => {
        const job = await prisma.job.findUnique({ where: { id: task.jobId } });
        if (!job) throw new Error('Job not found');
        
        // 获取用户画像（简化示例）
        const userProfile = { 
          resumeSummary: '5年Go后端', 
          preferences: '云原生/高并发', 
          expectedSalary: '30-50K' 
        };
        
        const analysis = await analyzeJob(job.jdContent, userProfile);
        if (!analysis) throw new Error('AI returned invalid output');
        
        // 4. 事务更新：AI结果 + 任务状态
        await prisma.transaction([
          prisma.job.update({
            where: { id: task.jobId },
            data: {
              aiScore: analysis.score,
              aiSummary: analysis.summary,
              aiRecommendation: analysis.recommendation,
              aiTechMatch: analysis.techStackMatch,
              aiRedFlags: analysis.redFlags,
              aiDimensions: analysis.dimensions,
              analyzedAt: new Date()
            }
          }),
          prisma.aiAnalysisTask.update({
            where: { id: task.id },
            data: { status: 'completed' }
          })
        ]);
      })
    );

    // 5. 处理失败任务
    const failures = results.filter(r => r.status === 'rejected');
    if (failures.length > 0) {
      const failedIds = tasks.slice(results.length - failures.length).map(t => t.id);
      await prisma.aiAnalysisTask.updateMany({
        where: { id: { in: failedIds } },
        data: { 
          status: 'pending', 
          retryCount: { increment: 1 },
          errorMsg: 'AI analysis failed'
        }
      });
    }

    return Response.json({ 
      processed: tasks.length, 
      succeeded: tasks.length - failures.length 
    });
    
  } catch (err) {
    console.error('Worker error:', err);
    return Response.json({ error: 'Worker crashed' }, { status: 500 });
  }
}

⚠️ 关键工程提醒

1.  Worker 触发方式选择：
    -   本地开发: 用 setInterval 在 Next.js 自定义 server 中触发，或单独写个 node worker.js 脚本。
    -   Vercel 部署: 使用 Vercel Cron Jobs (vercel.json 配置)，免费额度足够个人使用。
    -   Docker 部署: 用 node-cron 或独立进程运行 Worker。
2.  AI 成本控制: 在 Worker 中加入全局速率限制（如每分钟最多 10 次 AI 调用），避免突发流量导致账单爆炸。
3.  Popup 构建工具: 推荐使用 @crxjs/vite-plugin 或 plasmo 框架，它们原生支持 React/Preact + HMR，比手写 Webpack 配置高效 10 倍。
4.  事务安全: Worker 中的“更新 Job + 更新 Task”必须在同一个 transaction 中，否则可能出现 AI 分析完成但任务状态未更新，导致重复分析浪费 Token。

二、 云端定时调度：Vercel Cron Jobs

当你的 Next.js 项目部署到 Vercel 后，我们不能依赖本地的 setInterval。Vercel 提供了原生的 Cron Jobs 功能，可以定时触发我们的 AI Worker API。

⚙️ 1. 配置 vercel.json

在项目根目录创建 vercel.json，配置每 10 分钟触发一次 Worker：

{
  "crons": [
    {
      "path": "/api/workers/ai-analyze",
      "schedule": "*/10 * * * *"
    }
  ]
}

 2. 保护 Worker API 路由

Vercel 的 Cron 触发时会带上特定的 Header，我们需要在 API 路由中校验，防止被外部恶意刷接口。

首先，在 Vercel 项目的 Environment Variables 中添加一个环境变量：
CRON_SECRET = your-super-secret-random-string

然后，修改 Worker 路由代码：

// app/api/workers/ai-analyze/route.ts
import { NextRequest, NextResponse } from 'next/server';
import { processAiQueue } from '@/services/ai-worker-service'; // 将之前的逻辑抽离到 service

export async function GET(request: NextRequest) {
  // 1. 校验 Vercel Cron Secret
  const authHeader = request.headers.get('authorization');
  if (authHeader !== `Bearer {process.env.CRON_SECRET}`) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
  }

  // 2. 执行队列处理逻辑
  try {
    const result = await processAiQueue();
    return NextResponse.json(result);
  } catch (error) {
    console.error('Cron job failed:', error);
    return NextResponse.json({ error: 'Internal Server Error' }, { status: 500 });
  }
}

// Vercel Cron 默认发 GET 请求，如果需要用 POST 可以在 vercel.json 中配置，但 GET 更常见
export const dynamic = 'force-dynamic'; // 禁用缓存
export const maxDuration = 60; // Vercel Hobby 计划最大执行时间 60s，Pro 计划 300s

三、 AI 成本控制：JD 哈希缓存策略

痛点：Boss直聘上很多岗位的 JD 是一模一样的（尤其是外包或批量发布的岗位），或者同一个岗位你第二天又刷到了。如果每次都调 AI，Token 费用会很高。

解决方案：对 JD 文本进行 SHA-256 哈希，建立分析结果缓存表。

️ 1. Prisma Schema 增加 Cache 表

// prisma/schema.prisma 追加
model AiAnalysisCache {
  id          String   @id @default(cuid())
  jdHash      String   @unique // JD 内容的 SHA-256 哈希
  aiScore     Int
  aiSummary   String   @db.VarChar(500)
  aiRecommendation String
  aiTechMatch String[]
  aiRedFlags  String[]
  aiDimensions Json
  
  hitCount    Int      @default(1) // 命中次数，用于统计
  createdAt   DateTime @default(now())
  
  @@index([jdHash])
  @@map("ai_analysis_cache")
}

🧠 2. Worker 逻辑改造 (查缓存 -> 调 AI -> 写缓存)

修改 services/ai-worker-service.ts 中的核心处理逻辑：

import crypto from 'crypto';
import { analyzeJob } from './ai-job-analyzer';
import { prisma } from '@/lib/prisma';

// 计算文本哈希
function hashText(text: string): string {
  return crypto.createHash('sha256').update(text).digest('hex');
}

export async function processSingleTask(task: any, job: any, userProfile: any) {
  const jdHash = hashText(job.jdContent);

  // 1. 尝试命中缓存
  const cachedResult = await prisma.aiAnalysisCache.findUnique({
    where: { jdHash }
  });

  let analysisResult;

  if (cachedResult) {
    // ✅ 命中缓存：复用结果，增加命中计数
    analysisResult = {
      score: cachedResult.aiScore,
      summary: cachedResult.aiSummary,
      recommendation: cachedResult.aiRecommendation,
      techStackMatch: cachedResult.aiTechMatch,
      redFlags: cachedResult.aiRedFlags,
      dimensions: cachedResult.aiDimensions as any
    };
    
    await prisma.aiAnalysisCache.update({
      where: { jdHash },
      data: { hitCount: { increment: 1 } }
    });
    
    console.log(`♻️ Cache hit for job: {job.title}`);
  } else {
    // ❌ 未命中缓存：调用 AI 并写入缓存
    analysisResult = await analyzeJob(job.jdContent, userProfile);
    if (!analysisResult) throw new Error('AI analysis failed');

    await prisma.aiAnalysisCache.create({
      data: {
        jdHash,
        aiScore: analysisResult.score,
        aiSummary: analysisResult.summary,
        aiRecommendation: analysisResult.recommendation,
        aiTechMatch: analysisResult.techStackMatch,
        aiRedFlags: analysisResult.redFlags,
        aiDimensions: analysisResult.dimensions
      }
    });
    
    console.log(`🧠 AI analyzed and cached for job: ${job.title}`);
  }

  // 2. 将结果更新到 Job 主表
  await prisma.job.update({
    where: { id: job.id },
    data: {
      aiScore: analysisResult.score,
      aiSummary: analysisResult.summary,
      aiRecommendation: analysisResult.recommendation,
      aiTechMatch: analysisResult.techStackMatch,
      aiRedFlags: analysisResult.redFlags,
      aiDimensions: analysisResult.dimensions,
      analyzedAt: new Date()
    }
  });
}
